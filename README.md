sleep 1000           # Задержка 1000 мс (1 секунда)

wait 500            # Аналогично sleep

delay 250           # Задержка 250 мс

pause 300           # Пауза 300 мс

mouse 100 200        # Переместить мышь на +100, +200 пикселей

move_mouse -50 30    # Переместить на -50, +30

cursor 0 150         # Аналогично mouse

absolute_mouse 500 300    # Переместить в координаты (500, 300)

abs_mouse 0 0             # В верхний левый угол

screen_mouse 1920 1080    # В правый нижний угол (для FullHD)

click left             # Левый клик

click right            # Правый клик

click middle           # Клик средней кнопкой
positioncheck          # Показать текущие координаты мыши

mouse_position         # Аналогично
